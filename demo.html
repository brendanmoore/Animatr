Animation with event listening and callbacks

    //Set a counter
    var i=0;

    var anim = new Animatr(document.querySelector('.test2'), {
      css: { background: 'blue',
             color: 'white' },
      iterationCount: 12,
      iterationCallback: function(e){ //Callback on completion of animation interation
        i++;
        this.innerHTML = 'Called ' + i + ' times';
      },
      startCallback: function(e){ //Callback on start of animation
        this.innerHTML = 'Animation Start';
      },
      endCallback: function(e){ //Callback on end of animation
        this.innerHTML = 'Animation End';
      }

    });
